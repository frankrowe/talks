D3 and GIS


// var us = US county features
var path = d3.geo.path()
d3.selectAll("g")
  .data(us)
  .attr("class", "counties")
  .enter()
  .append("path")
  .attr("d", path)

D3 and GIS


//map unemployment rates to css classes
var quantize = d3.scale.quantize()
  .domain([0, .15])
  .range(d3.range(9).map(function(i) {
    return "q" + i
  }))

quantize(0.04) // q1
quantize(0.08) // q4
quantize(0.15) // q8
